CREATE TABLE "User" (
    id_user SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,  
    name VARCHAR(100),
    age INTEGER,
    apiKey VARCHAR(255) UNIQUE,
    imageProfile TEXT,  
    amount DECIMAL(15, 2) DEFAULT 0.00,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()  
);
